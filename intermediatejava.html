import java.util.Scanner;
class ExceptionExample {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter first number: ");
        int a = sc.nextInt();
        System.out.print("Enter second number: ");
        int b = sc.nextInt();
        try {
            int result = a / b;  // may throw ArithmeticException
            System.out.println("Result: " + result);
        } 
        catch (ArithmeticException e) {
            System.out.println("Exception caught: Division by zero is not allowed.");
        }
        System.out.println("Program continues after exception handling.");       
        sc.close();
    }
}

import java.io.*;
class FileReaderExample {
    // Method that reads a file and throws an exception if not found
    static void readFile(String fileName) throws FileNotFoundException {
        FileReader file = new FileReader(fileName);
        BufferedReader reader = new BufferedReader(file);
        try {
            String line;
            System.out.println("File content:");
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
        } 
        catch (IOException e) {
            System.out.println("Error reading the file.");
        } 
        finally {
            try {
                reader.close();
            } catch (IOException e) {
                System.out.println("Error closing the file.");
            }
        }
    }
    public static void main(String[] args) {
        try {
            // Try to read a file named "data.txt"
            readFile("data.txt");
        } 
        catch (FileNotFoundException e) {
            System.out.println("Exception caught: File not found!");
        }
    }
}

import java.io.*;
import java.util.*;
class PositiveNumberException extends Exception {
    public PositiveNumberException(String message) {
        super(message);
    }
}
class ReadNumbersFromFile {
    // Method to read numbers and throw exception if any number is positive
    static void readNumbers(String fileName) throws IOException, PositiveNumberException {
        File file = new File(fileName);
        Scanner sc = new Scanner(file);
        while (sc.hasNext()) {
            if (sc.hasNextInt()) {
                int num = sc.nextInt();
                if (num > 0) {
                    sc.close();
                    throw new PositiveNumberException("Positive number found: " + num);
                }
            } else {
                sc.next(); // skip non-integer tokens
            }
        }
        System.out.println("All numbers are non-positive.");
        sc.close();
    }
    public static void main(String[] args) {
        try {
            // Try to read from "numbers.txt"
            readNumbers("numbers.txt");
        } 
        catch (FileNotFoundException e) {
            System.out.println("Error: File not found!");
        } 
        catch (PositiveNumberException e) {
            System.out.println("Exception caught: " + e.getMessage());
        } 
        catch (IOException e) {
            System.out.println("An input/output error occurred.");
        }
    }
}

Output:

Exception caught: Positive number found: 3

import java.io.*;

class EmptyFileException extends Exception {
    public EmptyFileException(String message) {
        super(message);
    }
}
class FileCheckExample {
    // Method to read file and throw exception if empty
    static void readFile(String fileName) throws IOException, EmptyFileException {
        File file = new File(fileName);
        if (!file.exists()) {
            throw new FileNotFoundException("File not found: " + fileName);
        }
        if (file.length() == 0) {
            throw new EmptyFileException("The file is empty.");
        }
        BufferedReader reader = new BufferedReader(new FileReader(file));
        String line;
        System.out.println("File content:");
        while ((line = reader.readLine()) != null) {
            System.out.println(line);
        }
        reader.close();
    }
    public static void main(String[] args) {
        try {
            // Try to read "data.txt"
            readFile("data.txt");
        } 
        catch (FileNotFoundException e) {
            System.out.println("Error: " + e.getMessage());
        } 
        catch (EmptyFileException e) {
            System.out.println("Exception caught: " + e.getMessage());
        } 
        catch (IOException e) {
            System.out.println("An I/O error occurred.");
        }
    }
}

Output:

Exception caught: The file is empty.

import java.util.*;
// Custom exception class
class DuplicateNumberException extends Exception {
    public DuplicateNumberException(String message) {
        super(message);
    }
}
class DuplicateCheckExample {
    // Method to check for duplicates
    static void checkDuplicates(List numbers) throws DuplicateNumberException {
        Set uniqueNumbers = new HashSet<>();
        for (int num : numbers) {
            if (!uniqueNumbers.add(num)) {  // add() returns false if num already exists
                throw new DuplicateNumberException("Duplicate number found: " + num);
            }
        }
        System.out.println("No duplicates found. All numbers are unique.");
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        List numbers = new ArrayList<>();
        System.out.print("Enter how many numbers you want to input: ");
        int count = sc.nextInt();
        System.out.println("Enter " + count + " integers:");
        for (int i = 0; i < count; i++) {
            numbers.add(sc.nextInt());
        }
        try {
            checkDuplicates(numbers);
        } 
        catch (DuplicateNumberException e) {
            System.out.println("Exception caught: " + e.getMessage());
        }
        sc.close();
    }
}

Output:

Enter how many numbers you want to input: 5
Enter 5 integers:
1 2 3 4 5
No duplicates found. All numbers are unique.

import java.util.*;
// Custom exception class
class NoVowelException extends Exception {
    public NoVowelException(String message) {
        super(message);
    }
}
class VowelCheckExample {
    // Method to check for vowels
    static void checkVowels(String text) throws NoVowelException {
        String lowerText = text.toLowerCase();
        boolean hasVowel = false;
        for (char ch : lowerText.toCharArray()) {
            if ("aeiou".indexOf(ch) != -1) {
                hasVowel = true;
                break;
            }
        }
        if (!hasVowel) {
            throw new NoVowelException("The string does not contain any vowels.");
        }
        System.out.println("The string contains vowels.");
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = sc.nextLine();
        try {
            checkVowels(input);
        } 
        catch (NoVowelException e) {
            System.out.println("Exception caught: " + e.getMessage());
        }
        sc.close();
    }
}

Output:

Enter a string: Hello
The string contains vowels.

import java.util.*;
class MultipleCatchExample {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        try {
            System.out.print("Enter an integer: ");
            int num = sc.nextInt();
            int result = 100 / num; // May throw ArithmeticException
            System.out.println("Result of 100 / " + num + " = " + result
            int[] arr = new int[3];
            arr[5] = num; // May throw ArrayIndexOutOfBoundsException
        }
        catch (ArithmeticException e) {
            System.out.println("ArithmeticException caught: Cannot divide by zero.");
        }
        catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("ArrayIndexOutOfBoundsException caught: Invalid array index.");
        }
        catch (InputMismatchException e) {
            System.out.println("InputMismatchException caught: Please enter a valid integer.");
        }
        System.out.println("Program continues after exception handling.");
        sc.close();
    }
}

Output:

Enter an integer: 0
ArithmeticException caught: Cannot divide by zero.
Program continues after exception handling.

import java.util.Scanner;
class NestedTryExample {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        try {
            System.out.print("Enter the size of the array: ");
            int size = sc.nextInt();
            int[] arr = new int[size];
            try {
                System.out.print("Enter an index to store value: ");
                int index = sc.nextInt();
                System.out.print("Enter a number to store at index " + index + ": ");
                int num = sc.nextInt();
                arr[index] = num; // May throw ArrayIndexOutOfBoundsException
                System.out.println("Value stored successfully at index " + index);
            } 
            catch (ArrayIndexOutOfBoundsException e) {
                System.out.println("Inner catch: Invalid array index.");
            }
            // Division operation in outer try
            System.out.print("Enter a number to divide 100 by: ");
            int divisor = sc.nextInt();
            int result = 100 / divisor; // May throw ArithmeticException
            System.out.println("Result: " + result);
        } 
        catch (ArithmeticException e) {
            System.out.println("Outer catch: Cannot divide by zero.");
        } 
        catch (InputMismatchException e) {
            System.out.println("Outer catch: Invalid input type.");
        }
        System.out.println("Program continues after nested try-catch blocks.");
        sc.close();
    }
}

Output:

Enter the size of the array: 3
Enter an index to store value: 5
Enter a number to store at index 5: 10
Inner catch: Invalid array index.
Enter a number to divide 100 by: 5
Result: 20
Program continues after nested try-catch blocks.

    SwitchFramesExample() {
        // First Frame
        f1 = new Frame("Frame 1");
        f1.setLayout(new FlowLayout());
        Button toF2 = new Button("Go to Frame 2");
        f1.add(toF2);
        f1.setSize(300, 200);
        f1.setVisible(true);

        // Second Frame
        f2 = new Frame("Frame 2");
        f2.setLayout(new FlowLayout());
        Button toF1 = new Button("Go to Frame 1");
        f2.add(toF1);
        f2.setSize(300, 200);

        // Button action to switch to Frame 2
        toF2.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                f1.setVisible(false);
                f2.setVisible(true);
            }
        });

        // Button action to switch to Frame 1
        toF1.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                f2.setVisible(false);
                f1.setVisible(true);
            }
        });

        // Window closing for Frame 1
        f1.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                f1.dispose();
            }
        });

        // Window closing for Frame 2
        f2.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                f2.dispose();
            }
        });
    }

    public static void main(String[] args) {
        new SwitchFramesExample();
    }
}

public class RectangleApplet extends Applet {
    public void paint(Graphics g) {
        // Draw a rectangle at (50,50) with width 150 and height 100
        g.drawRect(50, 50, 150, 100);
    }
}

Output:

A rectangle is drawn at coordinates (50,50) with width 150 and height 100.

import java.applet.Applet;
import java.awt.Graphics;

public class ConcentricCirclesApplet extends Applet {
    public void paint(Graphics g) {
        int x = 100; // center x
        int y = 100; // center y
        int radius = 20; // starting radius

        // Draw 4 concentric circles
        for(int i = 1; i <= 4; i++) {
            g.drawOval(x - radius, y - radius, 2 * radius, 2 * radius);
            radius += 20; // increase radius for next circle
        }
    }
}

Output:

Four concentric circles are drawn with increasing radii centered at (100,100).
